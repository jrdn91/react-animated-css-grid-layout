## Install

```bash
npm install --save react-css-grid-layout
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-css-grid-layout'
import 'react-css-grid-layout/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
